\section{Round-3 Objective}
\label{sec:round3-objective}
\textbf{Path pursued: (C) obstruction/correction (sharpening the triangular obstruction).}

Round~2 reduced (Q2) to the existence of an ``almost triangular Golomb ruler'' family by proving a per-difference multiplicity bound
\( |P_d(S;M)|\le E_M+1 \) (Proposition~\ref{prop:triangular-golomb-obstruction}(2)), where $E_M:=|[1,M]\setminus B|$.

In this round I strengthen that obstruction in a way that is \emph{strictly sharper and more structural}:
for each fixed difference $d$, \emph{all} pairwise sums of starting points in $P_d$ (shifted by $d$) are forced to lie in the bad set.
Using a sharp lower bound for restricted sumsets in $\mathbb Z$, this yields an improved multiplicity bound
\[
|P_d(S;M)|\le \frac{E_M+3}{2},
\]
improving the Round~2 constant by a factor $\approx 2$.

\section{Round-2 Foundation Used}
\label{sec:round2-foundation-used}
I use the following Round~2 results (as established in \texttt{14\_v2.tex}):
\begin{itemize}
\item Lemma~\ref{lem:parallelogram} (parallelogram collision): if $a,a+d,c,c+d\in S$ with $a<c$ and $2a+d,2c+d\le M$, then $a+c+d\in\mathrm{Bad}(S;M)$.
\item The definitions $P_d(S;M)=\{a\in S: a+d\in S,\ 2a+d\le M\}$ and $E_M=|\mathrm{Bad}(S;M)|$.
\end{itemize}

\section{New Insight / Tool (Round-3)}
\label{sec:round3-new-tool}
The new tool is a \emph{clique amplification} of Lemma~\ref{lem:parallelogram}:
for a fixed $d$, \emph{every} pair of distinct starting points $a<c$ in $P_d$ generates a bad sum $a+c+d$.
Thus the bad set contains a translate of the \emph{restricted sumset}
\[
P_d\oplus P_d:=\{a+c:\ a,c\in P_d,\ a<c\}.
\]
The key additive-combinatorial input is an elementary sharp bound in $\mathbb Z$:
if $X\subset\mathbb Z$ has $|X|=k\ge 2$, then $|X\oplus X|\ge 2k-3$.

\section{Attack Plan (Round-3)}
\label{sec:round3-attack-plan}
\textbf{Gap after Round~2.}
Round~2 gave a per-difference bound $|P_d|\le E_M+1$, obtained by extracting only $|P_d|-1$ bad sums from the difference class $d$.

\textbf{Round~3 goal.}
Exploit \emph{all} pairs in $P_d$:
\begin{enumerate}
\item show $\mathrm{Bad}(S;M)$ contains $(P_d\oplus P_d)+d$ for every $d$;
\item lower bound $|P_d\oplus P_d|$ sharply by $2|P_d|-3$;
\item deduce the improved per-difference bound $|P_d|\le (E_M+3)/2$.
\end{enumerate}
This adds a genuinely new, checkable constraint to the ``triangular Golomb'' reduction.

\section{Work (Round-3)}
\label{sec:round3-work}
We keep the Round~2 notation: $M\ge 2$, $S\subseteq[1,M]$, $r_S(n)$ counts unordered representations with $a\le b$, and
$\mathrm{Bad}(S;M)=[1,M]\setminus B(S)$.

\subsection{From a fixed difference class to a translate of a restricted sumset}
\label{subsec:translate-restricted-sumset}
\begin{lemma}[Difference class $d$ forces a translate of $P_d\oplus P_d$ into $\mathrm{Bad}$]
\label{lem:pd-oplus-in-bad}
Let $M\ge 2$ and $S\subseteq[1,M]$. Fix $d\ge 1$ and define
\[
P_d:=P_d(S;M)=\{a\in S: a+d\in S\ \text{and}\ 2a+d\le M\}.
\]
Then for every pair $a<c$ in $P_d$ one has
\[
 a+c+d\in\mathrm{Bad}(S;M).
\]
Equivalently,
\begin{equation}
\label{eq:translate-oplus-subset-bad}
 (P_d\oplus P_d)+d\ \subseteq\ \mathrm{Bad}(S;M),
\end{equation}
where $P_d\oplus P_d:=\{a+c:\ a,c\in P_d,\ a<c\}$.
\end{lemma}

\begin{proof}
Let $a<c$ be in $P_d$. Then $a,a+d,c,c+d\in S$ and $2a+d,2c+d\le M$ by definition of $P_d$.
Applying Lemma~\ref{lem:parallelogram} (Round~2) gives $a+c+d\in\mathrm{Bad}(S;M)$.
This holds for every such pair $a<c$, yielding \eqref{eq:translate-oplus-subset-bad}.
\end{proof}

\subsection{A sharp lower bound for restricted sumsets in $\mathbb Z$}
\label{subsec:restricted-sumset-bound}
\begin{lemma}[Elementary restricted sumset bound in $\mathbb Z$]
\label{lem:restricted-sumset-2k-3}
Let $X\subseteq\mathbb Z$ be finite with $|X|=k\ge 2$. Define the restricted sumset
\[
X\oplus X:=\{x_i+x_j:\ x_i,x_j\in X,\ x_i<x_j\}.
\]
Then
\begin{equation}
\label{eq:restricted-sumset-lb}
 |X\oplus X|\ge 2k-3.
\end{equation}
\end{lemma}

\begin{proof}
Write $X=\{x_1<\cdots<x_k\}$.
Consider the $k-1$ sums
\[
 x_1+x_2 < x_1+x_3 < \cdots < x_1+x_k
\]
and the $k-2$ sums
\[
 x_2+x_k < x_3+x_k < \cdots < x_{k-1}+x_k.
\]
All these $2k-3$ sums belong to $X\oplus X$.
They are pairwise distinct: the first chain is strictly increasing; the second chain is strictly increasing; and every sum in the first chain is $\le x_1+x_k < x_2+x_k$, so the two chains are disjoint.
Thus $|X\oplus X|\ge 2k-3$.
\end{proof}

\begin{remark}[On the sharpness of \eqref{eq:restricted-sumset-lb}]
The bound \eqref{eq:restricted-sumset-lb} is sharp (e.g.\ for arithmetic progressions), but the equality case is \emph{not} unique in $\mathbb Z$.
For instance $X=\{0,1,3,4\}$ has $|X|=4$ and
\[
X\oplus X=\{1,3,4,5,7\},
\]
so $|X\oplus X|=5=2|X|-3$ even though $X$ is not an arithmetic progression.
Accordingly, in this round we use only the lower bound \eqref{eq:restricted-sumset-lb}.
\end{remark}

\subsection{A sharpened per-difference obstruction}
\label{subsec:sharpened-per-d}
\begin{proposition}[Sharpened per-difference bound]
\label{prop:sharpened-per-d}
Let $M\ge 2$, $S\subseteq[1,M]$, and $E_M:=|\mathrm{Bad}(S;M)|$.
For every $d\ge 1$ let $P_d=P_d(S;M)$.
Then for all $d\ge 1$,
\begin{equation}
\label{eq:improved-per-d}
 |P_d|\le \frac{E_M+3}{2}.
\end{equation}
\end{proposition}

\begin{proof}
If $|P_d|\le 1$ then \eqref{eq:improved-per-d} is trivial.
Assume $k:=|P_d|\ge 2$.
By Lemma~\ref{lem:pd-oplus-in-bad},
$(P_d\oplus P_d)+d\subseteq\mathrm{Bad}(S;M)$, hence
\[
 E_M\ge |(P_d\oplus P_d)+d| = |P_d\oplus P_d|.
\]
Applying Lemma~\ref{lem:restricted-sumset-2k-3} gives $|P_d\oplus P_d|\ge 2k-3$, so $E_M\ge 2k-3$, i.e.\ $k\le (E_M+3)/2$.
\end{proof}

\begin{remark}[Relation to Round~2 bounds]
Round~2 obtained $|P_d|\le E_M+1$ from a single anchored family of bad sums.
Proposition~\ref{prop:sharpened-per-d} strengthens this to roughly half the allowance by exploiting \emph{all} pairs $a<c$ in $P_d$ via the translate containment \eqref{eq:translate-oplus-subset-bad}.
\end{remark}

\section{Adversarial Verification}
\label{sec:round3-adversarial}
I stress-test the new claims.
\begin{itemize}
\item \textbf{Boundary $|P_d|=0,1$.} Lemma~\ref{lem:pd-oplus-in-bad} is vacuous and Proposition~\ref{prop:sharpened-per-d} gives $|P_d|\le (E_M+3)/2$, automatic.

\item \textbf{Range of forced bad sums.}
For $a<c\in P_d$, we used Lemma~\ref{lem:parallelogram}, which already verifies $a+c+d\le M$ under the triangle hypotheses. Here those hypotheses are built into $P_d$, so there is no hidden assumption.

\item \textbf{Potential collapse of many pairs into few sums.}
The map $(a,c)\mapsto a+c$ is not injective, so one cannot deduce $|P_d\oplus P_d|=\binom{|P_d|}{2}$.
Lemma~\ref{lem:restricted-sumset-2k-3} is exactly the tool that controls this collapse from below.

\item \textbf{Compatibility with Round~2.}
The new bound implies the Round~2 bound (since $(E_M+3)/2\le E_M+1$ for all $E_M\ge 1$), so no contradiction arises.
\end{itemize}

\section{Final}
\label{sec:round3-final}
\textbf{UNRESOLVED (BUT STRICTLY ADVANCED).}

Round~3 strengthens the Round~2 triangular obstruction:
for each $d\ge 1$ the bad set contains $(P_d\oplus P_d)+d$, hence
\[
E_M\ge |P_d\oplus P_d|\ge 2|P_d|-3,
\]
so the per-difference multiplicity bound improves from
$|P_d|\le E_M+1$ to $|P_d|\le (E_M+3)/2$.

The principal open gap remains: determine whether there exists an infinite nested family $A\subseteq\mathbb N$ for which $E_M=o(\sqrt M)$, equivalently whether the cumulative constraints on all difference classes $P_d(A\cap[1,M];M)$ can be met simultaneously for all large $M$.
