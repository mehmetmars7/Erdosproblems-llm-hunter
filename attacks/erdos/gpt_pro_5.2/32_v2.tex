\section{Round 2: Continuation \& Gap-Closure (Erd\H{o}s Problem \#32)}

\subsection{1) ROUND-2 OBJECTIVE}
\textbf{Path (C): obstruction / correction.}

Round-1 identified that (i) elementary finite obstructions (CRT and factorial blocks) do not by themselves yield the asymptotic lower bounds needed to settle the $o((\log N)^2)$ / $O(\log N)$ questions for the \emph{actual primes}, and (ii) any successful improvement must exploit structure beyond mere density.
In this round we prove a rigorous \emph{strategy barrier}: there exist \emph{prime-like} sets $\mathcal P'\subseteq\mathbb N$ with counting function $\mathcal P'(x)\sim x/\log x$ for which \emph{every} additive complement $B$ must satisfy
\[
\liminf_{x\to\infty}\frac{B(x)}{(\log x)^2}\ge 1.
\]
Consequently, any approach that uses only ``PNT-scale'' information (growth like $x/\log x$) cannot prove an $o((\log N)^2)$ complement for the actual primes; it would falsely apply to such $\mathcal P'$.

\subsection{2) ROUND-1 FOUNDATION USED}
We rely on the following Round-1 items (used only as foundations, not reproved here):
\begin{enumerate}
\item Definition: $A$ is an additive complement to primes if $\exists N_0$ such that every $n\ge N_0$ satisfies $n=p+a$ with $p\in\mathcal P$, $a\in A$.
\item Finite truncation principle (implicit in the Round-1 set-cover viewpoint): to cover $[M,x]$ by a sumset $P'+B$, it suffices to consider $B\cap[0,x]$, since if $p'+b\le x$ then $b\le x$.
\end{enumerate}

\subsection{3) NEW INSIGHT / TOOL (ROUND-2)}
New input is a \emph{hard-to-complement} theorem (Kolountzakis, 1996): for a suitable random prime-like set $\mathcal P'$, \emph{every} additive complement $B$ must have size $\gtrsim(\log x)^2$. The proof uses:
\begin{enumerate}
\item a packing/greedy construction producing many \emph{disjoint truncated translates} of $B$;
\item independence of membership events for $\mathcal P'$ on disjoint sets;
\item a union bound over all small $B\cap[0,x]$ and a Borel--Cantelli argument.
\end{enumerate}

\subsection{4) ATTACK PLAN (ROUND-2)}
\textbf{Round-1 gap addressed:} show why arguments based only on coarse density information cannot improve the $(\log N)^2$ scale.

\textbf{Plan:} define a random set $\mathcal P'$ with inclusion probabilities $\mathbb P(n\in\mathcal P')=1/\log n$ (for $n\ge4$). Show that almost surely:
\begin{enumerate}
\item $\mathcal P'(x)\sim x/\log x$ (so $\mathcal P'$ is prime-like in growth);
\item if $B(x)\le C(\log x)^2$ infinitely often with some $C<1$, then $B$ cannot be an additive complement of $\mathcal P'$.
\end{enumerate}
Thus, almost surely every additive complement satisfies $\liminf B(x)/(\log x)^2\ge1$.

\subsection{5) WORK (ROUND-2)}

\subsubsection*{5.1 Setup}
Let $\mathbb N=\{0,1,2,\dots\}$. Define a random set $\mathcal P'\subseteq\{4,5,6,\dots\}$ by choosing each integer $n\ge4$ independently with probability
\[
\mathbb P(n\in\mathcal P')=\frac1{\log n}.
\]
For $B\subseteq\mathbb N$ define $B(x):=|B\cap[0,x]|$.

We say $B$ is an \emph{additive complement} of $\mathcal P'$ if $\exists N_0$ such that every $n\ge N_0$ can be written $n=p'+b$ with $p'\in\mathcal P'$, $b\in B$.

\subsubsection*{5.2 Main theorem (barrier)}
\textbf{Theorem.}
With probability $1$ (over the choice of $\mathcal P'$), every additive complement $B$ of $\mathcal P'$ satisfies
\[
\liminf_{x\to\infty}\frac{B(x)}{(\log x)^2}\ge 1.
\]
In particular, for every fixed $C<1$ there is almost surely no additive complement $B$ such that $B(x)\le C(\log x)^2$ for infinitely many $x$.

\subsubsection*{5.3 Combinatorial packing lemma}
Fix $x\in\mathbb N$ and finite $B\subseteq[0,x]$ with $m:=|B|=B(x)$.
For $a\in[0,x]$ define $B_{\le a}:=B\cap[0,a]$ and
\[
a-B_{\le a}:=\{a-b:\ b\in B_{\le a}\}.
\]

Fix $\lambda\in(0,1)$ and set
\[
s:=\left\lfloor \frac{\lambda x}{m}\right\rfloor.
\]

\textbf{Lemma 1 (many disjoint truncated translates).}
There exist integers $a_1,\dots,a_n\in[0,x]$ such that:
\begin{enumerate}
\item the sets $a_j-B_{\le a_j}$ are pairwise disjoint;
\item each $(a_j-B_{\le a_j})\cap[1,s]=\emptyset$;
\item and
\[
n\ge \frac{(1-\lambda)x}{m^2}.
\]
\end{enumerate}

\textbf{Proof.}
Choose a maximal family $a_1,\dots,a_n$ satisfying (1)--(2). Let
\[
G:=[1,s]\ \cup\ \bigcup_{j=1}^n (a_j-B_{\le a_j}).
\]
By maximality, for every $y\in[0,x]$ the set $y-B_{\le y}$ intersects $G$, so $y\in G+B$.
Thus $[0,x]\subseteq G+B$ and hence
\[
x+1 \le |G+B|\le |G|\cdot|B|\le (s+nm)m.
\]
Rearranging gives $n\ge (x+1)/m^2 - s/m \ge x/m^2 - s/m \ge (1-\lambda)x/m^2$ since $s\le \lambda x/m$.
\qed
