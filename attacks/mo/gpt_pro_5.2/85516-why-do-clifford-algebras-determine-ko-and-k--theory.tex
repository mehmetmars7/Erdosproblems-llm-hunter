\section{MO 85516: Why do Clifford algebras determine KO and K-theory?}

\subsection*{1) FORMAL RESTATEMENT}

\paragraph{Definitions.}
Fix $k\ge 0$.
Let $C_k$ denote the real Clifford algebra on $\mathbb{R}^k$ with \emph{negative definite} quadratic form; it is $\mathbb{Z}/2$-graded by parity.
Let $\mathcal{M}_k$ be the exact category of finite-dimensional $\mathbb{Z}/2$-graded left $C_k$-modules, and let $M_k$ be its Grothendieck group.
Restriction along the canonical inclusion $C_k\hookrightarrow C_{k+1}$ induces a group homomorphism
\[
\mathrm{res}\colon M_{k+1}\to M_k.
\]
Define
\[
A_k := M_k/\mathrm{im}(\mathrm{res}),
\]
which matches the “$M_k/M_{k+1}$” quotient appearing in the ABS construction.

Let $\widetilde{KO}(S^k)\cong KO^{-k}(*)$ be reduced real $K$-theory of the $k$-sphere, canonically identified with the $(-k)$-th coefficient group of $KO$.

\paragraph{Corrected statement to prove.}
\begin{theorem}[ABS isomorphism, via Fredholm model]\label{thm:ABS-via-Fredholm}
For each $k\ge 0$ there is a natural isomorphism
\[
A_k \;\cong\; KO^{-k}(*)\;\cong\;\widetilde{KO}(S^k),
\]
and under this identification the ABS map is the isomorphism. Consequently,
\[
\bigoplus_{k\ge 0} A_k \;\cong\; KO^{-\bullet}(*)
\]
as graded rings, where multiplication on the left is induced by graded tensor product of Clifford modules.
\end{theorem}

\subsection*{2) QUICK LITERATURE/CONTEXT CHECK (web available)}

A direct operator-theoretic proof appears in Atiyah--Singer, \emph{Index theory for skew-adjoint Fredholm operators}. They:
\begin{itemize}[leftmargin=2em]
\item define $A_k$ as the Grothendieck group of graded $C_k$-modules modulo those extendable to $C_{k+1}$,
\item define a space $O^{-k}$ of skew-adjoint Fredholm operators with $C_k$-symmetry,
\item construct an index map $\mathrm{ind}_k\colon O^{-k}\to A_k$ inducing a bijection $\pi_0(O^{-k})\cong A_k$,
\item identify $A_k$ with $KR^{-k}(\mathrm{pt})$, and in particular with $KO^{-k}(\mathrm{pt})$ in the untwisted real case,
\item and compare this isomorphism with the ABS map.
\end{itemize}

\item Prove (using their results) that connected components of the relevant Fredholm space are classified by the “Clifford-module index” in $A_k$.
\item Identify this component group with $KO^{-k}(*)$ and note the constructed isomorphism is the ABS map.
\end{enumerate}

\subsection*{4) WORK (proof, with explicit reliance on stated theorems)}

\subsubsection*{Step 1: Identify $A_k$ with the quotient of Clifford module Grothendieck groups}

\begin{lemma}\label{lem:Ak-is-cokernel}
With $A_k:=M_k/\mathrm{im}(M_{k+1}\to M_k)$ as above, $A_k$ is exactly the group described by Atiyah--Singer as “the Grothendieck group of graded $C_k$-modules, modulo those extendable to $C_{k+1}$-modules”.
\end{lemma}

\begin{proof}
A graded $C_{k+1}$-module restricts to a graded $C_k$-module by forgetting the last generator.
The subgroup of $M_k$ generated by such restrictions is precisely $\mathrm{im}(M_{k+1}\to M_k)$.
Quotienting $M_k$ by this subgroup yields $A_k$.
\end{proof}

\subsubsection*{Step 2: The skew-adjoint Fredholm space and its index}

\paragraph{The operator space.}
Let $\mathcal{H}$ be a separable real $\mathbb{Z}/2$-graded Hilbert space equipped with a representation of $C_k$ by bounded operators, such that each irreducible graded $C_k$-module occurs with infinite multiplicity (“ample”).
Let $O^{-k}$ denote the space of bounded \emph{degree-$1$} operators $B\colon\mathcal{H}\to\mathcal{H}$ that are
\begin{itemize}[leftmargin=2em]
\item skew-adjoint: $B^*=-B$,
\item Fredholm,
\item and $C_k$-linear in the graded sense (equivalently: $B$ anticommutes with the $C_k$ generators).
\end{itemize}

\begin{definition}[Clifford-module index]\label{def:indk}
Define
\[
\mathrm{ind}_k(B) := [\ker(B)]\in A_k,
\]
where $\ker(B)$ is finite-dimensional and inherits a graded $C_k$-module structure.
\end{definition}

\begin{proposition}[Atiyah--Singer]\label{prop:AS-5.1}
The map $\mathrm{ind}_k\colon O^{-k}\to A_k$ is locally constant and induces a bijection
\[
\pi_0(O^{-k}) \;\xrightarrow{\ \cong\ }\; A_k.
\]
\end{proposition}

\begin{proof}[Proof sketch with explicit dependencies]
Atiyah--Singer prove this as their Proposition 5.1.
The key points are:
\begin{itemize}[leftmargin=2em]
\item $\ker(B)$ is finite-dimensional and $C_k$-stable because $B$ is Fredholm and $C_k$-linear.
\item Under a continuous path $B_t$ in $O^{-k}$, the kernel can change only by “creation/annihilation” of finite-dimensional summands that extend to a $C_{k+1}$-module; this is exactly why the target is the quotient group $A_k$.
\item Surjectivity: given a finite-dimensional graded $C_k$-module $V$, embed $V$ as a $C_k$-submodule of $\mathcal{H}$ and define $B$ to be $0$ on $V$ and invertible on its orthogonal complement, arranged to satisfy the $C_k$-skew-adjoint conditions; then $\ker(B)\cong V$.
\item Injectivity: if two operators have the same index in $A_k$, one stabilizes by adding a trivial $C_{k+1}$-extendable summand to make the kernels isomorphic as $C_k$-modules and then constructs a path of operators through $O^{-k}$ connecting them.
\end{itemize}
These steps are carried out rigorously in Atiyah--Singer.
\end{proof}

\subsubsection*{Step 3: Identify $\pi_0(O^{-k})$ with $KO^{-k}(*)$}

\begin{proposition}[Fredholm model for $KO$ with Clifford symmetry]\label{prop:Fredholm-represents-KO}
There is a natural identification
\[
KO^{-k}(*) \;\cong\; \pi_0(O^{-k}),
\]
and more generally $KO^{-k}(X)$ is represented by homotopy classes of maps $X\to O^{-k}$.
\end{proposition}

\begin{proof}
This is part of the Atiyah--Singer framework: they identify $O^{-k}$ as the classifying space for (a real form of) $K$-theory with the appropriate Clifford symmetry, i.e.\ $KR^{-k}$, and for the point this gives $KR^{-k}(\mathrm{pt})\cong \pi_0(O^{-k})$.
In the untwisted real case $KR^{-k}(\mathrm{pt})\cong KO^{-k}(\mathrm{pt})$.
\end{proof}

\subsubsection*{Step 4: Conclude the ABS isomorphism}

\begin{proof}[Proof of Theorem \ref{thm:ABS-via-Fredholm}]
By Proposition \ref{prop:AS-5.1}, $\pi_0(O^{-k})\cong A_k$ via the index map $\mathrm{ind}_k$.
By Proposition \ref{prop:Fredholm-represents-KO}, $\pi_0(O^{-k})\cong KO^{-k}(*)$.
Composing gives a canonical isomorphism
\[
A_k \;\cong\; KO^{-k}(*).
\]
Identifying $KO^{-k}(*)\cong \widetilde{KO}(S^k)$ gives the desired isomorphism.

Finally, Atiyah--Singer compare this isomorphism with the Atiyah--Bott--Shapiro construction and show they coincide (their Proposition 5.2), hence the ABS map is an isomorphism.
\end{proof}

\subsection*{5) VERIFICATION}

\begin{itemize}[leftmargin=2em]
\item The quotient by extendable modules is essential: along a homotopy, kernel changes by adding/removing pairs corresponding to $C_{k+1}$-extensions.
\item Compactness issues are avoided by working in an ample Hilbert $C_k$-module $\mathcal{H}$ so stabilization is always possible.
\item The proof is “direct” in the sense that it identifies $KO$ with an operator space whose components are classified by Clifford-module indices. Bott periodicity is encoded in the Clifford-algebra periodicity and the homotopy type of $O^{-k}$.
\end{itemize}

\subsection*{6) FINAL}

\textbf{PROOF.}

\noindent\textbf{Clean theorem statement.}
The ABS quotient group $A_k=M_k/\mathrm{im}(M_{k+1}\to M_k)$ is naturally isomorphic to $KO^{-k}(*)\cong \widetilde{KO}(S^k)$, and the ABS map is the isomorphism (Theorem \ref{thm:ABS-via-Fredholm}).
