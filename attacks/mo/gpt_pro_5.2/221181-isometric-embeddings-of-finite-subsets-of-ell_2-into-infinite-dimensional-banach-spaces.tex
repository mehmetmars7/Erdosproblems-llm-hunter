\section{Isometric embeddings of finite subsets of $\ell_2$ into infinite-dimensional Banach spaces (MO 221181)}

\subsection*{1) FORMAL RESTATEMENT}
\paragraph{Literal statement is interrogative.} The ``statement'' is a yes/no question.

\paragraph{Conventions.}
\begin{itemize}[leftmargin=2em]
\item Scalars are $\RR$ unless stated otherwise; all norms are real norms. (The complex case is analogous.)
\item $\ell_2$ is the real separable Hilbert space with norm $\|\cdot\|_2$.
\item A Banach space means complete normed vector space.
\item A map $\phi:F\to X$ (with $F$ a metric space with metric $d_F$ and $X$ normed) is an \emph{isometric embedding} if for all $u,v\in F$,
\[\|\phi(u)-\phi(v)\|_X = d_F(u,v).\]
\end{itemize}

\paragraph{Precise decision problem.}
\begin{quote}
Determine whether
\[\exists\,\text{finite }F\subset \ell_2\ \exists\,\text{infinite-dimensional Banach }X\ \forall\,\phi:F\to X,\ \exists u,v\in F\text{ with }\|\phi(u)-\phi(v)\|_X\neq\|u-v\|_2.\]
\end{quote}

\paragraph{Stress points.}
\begin{itemize}[leftmargin=2em]
\item Small $|F|$ (especially $|F|\le 3$) where realization is flexible.
\item Whether $X$ may be assumed separable / reflexive / uniformly convex, etc. (Not stated.)
\item Whether ``Banach space'' is over $\RR$ or $\CC$.
\end{itemize}

\subsection*{2) QUICK LITERATURE/CONTEXT CHECK (browsing available)}
This problem is explicitly recorded as open in later work. For example, Kilbane's paper ``There is no finitely isometric Krivine's theorem'' lists the question as ``Problem 1.3'' and states ``This problem remains open'' (see the statement of Problem 1.3 in \href{https://arxiv.org/pdf/1708.01570}{arXiv:1708.01570}).

Kilbane's earlier paper \href{https://arxiv.org/abs/1609.08971}{arXiv:1609.08971} studies the question and provides partial positive results for certain classes (e.g. ultrametrics, and additional ``genericity'' statements in some settings). A 2022 overview on Dvoretzky-type results for \emph{locally finite} subsets of $\ell_2$ explicitly distinguishes the finite isometric question as open while noting negative results for certain \emph{infinite} locally finite subsets (see \href{https://arxiv.org/abs/2203.00166}{arXiv:2203.00166}, introduction).

\subsection*{3) ATTACK PLAN}
\paragraph{Proof track (show the answer is ``no counterexample exists'').}
\begin{enumerate}[leftmargin=2em]
\item Try to realize arbitrary finite Euclidean metrics inside arbitrary infinite-dimensional Banach spaces via finite-dimensional subspace selection + continuity/connectedness arguments.
\item Reduce to realizability of Euclidean simplices by prescribed edge lengths in normed spaces (Minkowski simplex realization).
\item Use ``finite determination'' or Dvoretzky-type ideas to pass from approximate Euclidean subspaces to exact finite configurations.
\end{enumerate}

\paragraph{Disproof track (construct a specific finite $F$ and $X$).}
\begin{enumerate}[leftmargin=2em]
\item Try smallest nontrivial candidate: $|F|=4$ (a nondegenerate Euclidean tetrahedron).
\item Seek $X$ with severe geometric restrictions on 3D subspaces (e.g. polyhedral norms with restricted sets of realizable tetrahedra).
\item Search for invariants of finite subsets preserved by isometries into $X$ but not satisfied by some $F\subset\ell_2$.
\end{enumerate}

\paragraph{Best path now.}
Given that the full decision problem is open, we pursue rigorous partial results, especially for smallest $|F|$.

\subsection*{4) WORK}

\subsubsection*{Phase 1: tiny cases}
We prove that every 3-point subset of $\ell_2$ embeds isometrically into any Banach space of dimension $\ge 2$.

\begin{lemma}[Any triangle metric embeds into any $\dim\ge 2$ normed space]
\label{lem:triangle}
Let $(X,\|\cdot\|)$ be a normed space with $\dim X\ge 2$. Let $a,b,c\ge 0$ satisfy the triangle inequalities
\[|a-b|\le c\le a+b.\]
Then there exist $u,v\in X$ such that $\|u\|=a$, $\|v\|=b$, and $\|u-v\|=c$.
\end{lemma}

\begin{proof}
If $a=0$, take $u=0$ and choose any $v$ with $\|v\|=b$; then $\|u-v\|=b=c$ by the inequalities. Similarly for $b=0$.
Assume $a,b>0$.

Pick any unit vector $x_0\in X$ and set $u:=a x_0$.
Since $\dim X\ge 2$, the unit sphere
\[S_X:=\{x\in X:\|x\|=1\}
\]
contains a continuous path from $x_0$ to $-x_0$.
Indeed, choose $y\in X$ linearly independent from $x_0$, and define a continuous map
\[\gamma(t)=\frac{(\cos\pi t)\,x_0+(\sin\pi t)\,y}{\|(\cos\pi t)\,x_0+(\sin\pi t)\,y\|}\in S_X,\qquad t\in[0,1],
\]
which satisfies $\gamma(0)=x_0$ and $\gamma(1)=-x_0$.

Define the continuous function $\varphi:[0,1]\to\RR$ by
\[\varphi(t):=\big\|u-b\gamma(t)\big\|.
\]
Then
\[\varphi(0)=\|a x_0-b x_0\|=|a-b|,\qquad \varphi(1)=\|a x_0+b x_0\|=a+b.
\]
Since $[0,1]$ is connected and $\varphi$ is continuous, $\varphi([0,1])$ is a connected subset of $\RR$ containing both endpoints; hence it contains the whole interval $[|a-b|,a+b]$.
Therefore there exists $t_\ast\in[0,1]$ with $\varphi(t_\ast)=c$.
Set $v:=b\gamma(t_\ast)$. Then $\|v\|=b$ and $\|u-v\|=\varphi(t_\ast)=c$.
\end{proof}

\begin{corollary}[3-point subsets of $\ell_2$ always embed]
\label{cor:3points}
Let $X$ be a Banach space with $\dim X\ge 2$. Then every 3-point metric space embeds isometrically into $X$. In particular, every 3-point subset $F\subset\ell_2$ admits an isometric embedding into $X$.
\end{corollary}

\begin{proof}
Let the metric space be $\{p_0,p_1,p_2\}$ with distances
\[a=d(p_0,p_1),\quad b=d(p_0,p_2),\quad c=d(p_1,p_2).
\]
By the triangle inequality, $|a-b|\le c\le a+b$.
Apply Lemma~\ref{lem:triangle} to obtain $u,v\in X$ with $\|u\|=a$, $\|v\|=b$, and $\|u-v\|=c$.
Define $\phi(p_0)=0$, $\phi(p_1)=u$, $\phi(p_2)=v$.
Then all pairwise distances are realized exactly.
\end{proof}

\subsubsection*{Phase 1: a universal embedding into $\ell_\infty^N$ (useful reduction)}
For completeness, we record the standard Fr\'echet (Kuratowski) embedding for finite metric spaces.

\begin{lemma}[Fr\'echet embedding into $\ell_\infty$]
\label{lem:frechet}
Let $(M,d)$ be a finite metric space with $|M|=n$. Fix an ordering $M=\{x_1,\dots,x_n\}$.
Define $\Phi:M\to\RR^n$ by
\[\Phi(x)=(d(x,x_1),\dots,d(x,x_n)).
\]
Then for all $x,y\in M$,
\[\|\Phi(x)-\Phi(y)\|_{\infty}=d(x,y).
\]
So $\Phi$ is an isometric embedding of $(M,d)$ into $\ell_\infty^n$.
\end{lemma}

\begin{proof}
For any index $i$ the triangle inequality gives
\[|d(x,x_i)-d(y,x_i)|\le d(x,y),\]
so $\|\Phi(x)-\Phi(y)\|_\infty\le d(x,y)$.
On the other hand, take $i$ such that $x_i=x$; then
\[|d(x,x)-d(y,x)|=d(x,y),
\]
so $\|\Phi(x)-\Phi(y)\|_\infty\ge d(x,y)$.
\end{proof}

Lemma~\ref{lem:frechet} implies: if a Banach space $X$ contains an isometric copy of $\ell_\infty^n$, then it contains isometrically every $n$-point metric space (in particular, every $n$-point subset of $\ell_2$). The challenge is that an arbitrary infinite-dimensional Banach space need not contain $\ell_\infty^n$ isometrically.

\subsection*{5) VERIFICATION}
\paragraph{Quantifiers and boundary cases.}
Lemma~\ref{lem:triangle} explicitly handles $a=0$ or $b=0$, and uses only $\dim X\ge 2$ for path-connectedness of the sphere. The constructed $\gamma(t)$ is well-defined because the numerator never vanishes: $x_0,y$ are linearly independent so $(\cos\pi t)x_0+(\sin\pi t)y\neq 0$ for all $t$.

\paragraph{Adversarial checks.}
The argument uses only continuity and connectedness, so it works for arbitrary norms (even non-strictly convex norms) and does not rely on differentiability.

\subsection*{6) FINAL}
\textbf{UNRESOLVED.}

\paragraph{(i) Strongest fully proved partial result obtained here.}
Every 3-point subset of $\ell_2$ embeds isometrically into every Banach space of dimension $\ge 2$ (Corollary~\ref{cor:3points}). Also every $n$-point metric space embeds isometrically into $\ell_\infty^n$ (Lemma~\ref{lem:frechet}).

\paragraph{(ii) First gap.}
We cannot decide whether there exists a 4-point subset $F\subset\ell_2$ and an infinite-dimensional Banach space $X$ such that $F$ fails to embed isometrically into $X$.

\paragraph{(iii) Top 3 next moves.}
\begin{enumerate}[leftmargin=2em]
\item Analyze realizability of a fixed nondegenerate Euclidean tetrahedron inside arbitrary 3D subspaces of candidate $X$ with strong geometric constraints (polyhedral, uniformly convex, etc.).
\item Search for an invariant of 4-point metrics preserved by isometric embeddings into $X$ (e.g. constraints on normed-space ``Cayley--Menger''-type determinants) that excludes some Euclidean tetrahedra.
\item Exploit known restrictions on isometric copies of $\ell_2^k$ or simplex structures in certain Banach spaces, attempting to force failure for some finite configuration.
\end{enumerate}

\paragraph{(iv) What a minimal counterexample might look like.}
If a counterexample exists, it must have $|F|\ge 4$ by Corollary~\ref{cor:3points}. A plausible minimal $F$ is a generic 4-point Euclidean tetrahedron and $X$ a Banach space whose 3D subspaces are far from Euclidean in a rigid (not merely quantitative) way.
